create_df <- function(string, no_vars){

#correlation across lists 
all_corr_mats <- function(etic_list){
  cross2(list_fs, etic_list) %>% 
   invoke_map(cor, .) %>% 
   map(., as_tibble, rownames = "cand_model") 
  
}
